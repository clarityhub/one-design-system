# ProgressBar

Used to display the linear progress of an action. Any children components will be rendered above the Progress Bar

```js
import ProgressBar from '@clarityhub/unity-web/lib/components/ProgressBar';
```

<Playground>
    {() => {
        class AnimatedProgress extends Component {
            constructor(props) {
                super(props);
                this.state = {
                    progress: parseInt(+new Date() / 100 % 100),
                };
            }

            componentDidMount() {
                this.timer = setInterval(() => {
                    this.setState({
                        progress: parseInt(+new Date() / 100 % 100),
                    })
                }, 1000);
            }

            componentWillUnmount() {
                if (this.timer) {
                    clearInterval(this.timer);
                }
            }

            render() {
                const { progress } = this.state;

                return (
                    <ProgressBar primary progress={progress} type="default">
                        <Badge type="primary">Uploading...</Badge>
                    </ProgressBar>
                );
            }
        }

        return <AnimatedProgress />;
    }}
</Playground>
